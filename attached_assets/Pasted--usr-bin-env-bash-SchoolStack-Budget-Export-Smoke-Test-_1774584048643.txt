#!/usr/bin/env bash
# SchoolStack Budget — Export Smoke Test (Complete)
#
# Tests ALL 14 export routes:
#   Public:  3 XLSX (POST, no auth)
#   Auth:    5 XLSX + 4 PDF + 2 JSON (GET, requires token)
#
# Each binary route tested through Netlify proxy AND direct to Railway.
#
# Usage:
#   ./scripts/smoke-test-exports.sh                        # Public only
#   ./scripts/smoke-test-exports.sh TOKEN MODEL_ID         # All routes
#
# Env overrides:
#   NETLIFY_URL   (default: https://budget.schoolstack.ai)
#   RAILWAY_URL   (default: https://schoolstackbudget.up.railway.app)

set -euo pipefail

TOKEN="${1:-}"
MODEL_ID="${2:-}"
NETLIFY_BASE="${NETLIFY_URL:-https://budget.schoolstack.ai}"
RAILWAY_BASE="${RAILWAY_URL:-https://schoolstackbudget.up.railway.app}"
OUT_DIR="/tmp/schoolstack-smoke-$(date +%s)"
mkdir -p "$OUT_DIR"

PASS=0; FAIL=0; SKIP=0

RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[0;33m'
CYAN='\033[0;36m'; BOLD='\033[1m'; NC='\033[0m'

# ── Validators ───────────────────────────────────────────────

is_html() { head -c 30 "$1" 2>/dev/null | grep -qi "<!doctype\|<html"; }

check_xlsx() {
  local file="$1" label="$2"
  if [ ! -f "$file" ]; then
    echo -e "  ${RED}❌ $label — no file${NC}"; FAIL=$((FAIL+1)); return; fi
  local size; size=$(wc -c < "$file" | tr -d ' ')
  if [ "$size" -lt 100 ]; then
    echo -e "  ${RED}❌ $label — ${size}B, likely error:${NC}"
    head -c 200 "$file"; echo; FAIL=$((FAIL+1)); return; fi
  if is_html "$file"; then
    echo -e "  ${RED}❌ $label — got HTML instead of XLSX (${size}B) — proxy serving SPA${NC}"
    FAIL=$((FAIL+1)); return; fi
  local magic; magic=$(od -A n -t x1 -N 4 "$file" | tr -d ' ')
  if [ "$magic" != "504b0304" ]; then
    echo -e "  ${RED}❌ $label — not ZIP (magic: $magic, ${size}B)${NC}"
    FAIL=$((FAIL+1)); return; fi
  local result; result=$(python3 -c "
import zipfile
try:
    z = zipfile.ZipFile('$file')
    bad = z.testzip()
    print(f'CORRUPT:{bad}' if bad else f'OK:{len(z.namelist())}')
except Exception as e:
    print(f'ERROR:{e}')
" 2>&1)
  local status="${result%%:*}" detail="${result#*:}"
  case "$status" in
    OK)      echo -e "  ${GREEN}✅ $label — valid (${size}B, ${detail} entries)${NC}"; PASS=$((PASS+1)) ;;
    CORRUPT) echo -e "  ${RED}❌ $label — corrupt: ${detail} (${size}B)${NC}"; FAIL=$((FAIL+1)) ;;
    ERROR)   echo -e "  ${RED}❌ $label — error: ${detail} (${size}B)${NC}"; FAIL=$((FAIL+1)) ;;
  esac
}

check_pdf() {
  local file="$1" label="$2"
  if [ ! -f "$file" ]; then
    echo -e "  ${RED}❌ $label — no file${NC}"; FAIL=$((FAIL+1)); return; fi
  local size; size=$(wc -c < "$file" | tr -d ' ')
  if [ "$size" -lt 100 ]; then
    echo -e "  ${RED}❌ $label — ${size}B${NC}"; FAIL=$((FAIL+1)); return; fi
  if is_html "$file"; then
    echo -e "  ${RED}❌ $label — got HTML instead of PDF (${size}B)${NC}"
    FAIL=$((FAIL+1)); return; fi
  if [ "$(head -c 5 "$file")" = "%PDF-" ]; then
    echo -e "  ${GREEN}✅ $label — valid PDF (${size}B)${NC}"; PASS=$((PASS+1))
  else
    echo -e "  ${RED}❌ $label — not PDF (${size}B)${NC}"; FAIL=$((FAIL+1)); fi
}

check_json() {
  local file="$1" label="$2"
  if [ ! -f "$file" ]; then
    echo -e "  ${RED}❌ $label — no file${NC}"; FAIL=$((FAIL+1)); return; fi
  local size; size=$(wc -c < "$file" | tr -d ' ')
  if [ "$size" -lt 10 ]; then
    echo -e "  ${RED}❌ $label — empty (${size}B)${NC}"; FAIL=$((FAIL+1)); return; fi
  if is_html "$file"; then
    echo -e "  ${RED}❌ $label — got HTML instead of JSON (${size}B)${NC}"
    FAIL=$((FAIL+1)); return; fi
  if python3 -c "import json; d=json.load(open('$file')); exit(1 if 'error' in d else 0)" 2>/dev/null; then
    echo -e "  ${GREEN}✅ $label — valid JSON (${size}B)${NC}"; PASS=$((PASS+1))
  elif python3 -c "import json; json.load(open('$file'))" 2>/dev/null; then
    local err; err=$(python3 -c "import json; print(json.load(open('$file')).get('error','?'))" 2>/dev/null)
    echo -e "  ${RED}❌ $label — error response: ${err}${NC}"; FAIL=$((FAIL+1))
  else
    echo -e "  ${RED}❌ $label — not JSON (${size}B)${NC}"; FAIL=$((FAIL+1))
  fi
}

compare_sizes() {
  local direct="$1" proxy="$2"
  [ ! -f "$direct" ] || [ ! -f "$proxy" ] && return
  local d; d=$(wc -c < "$direct" | tr -d ' ')
  local p; p=$(wc -c < "$proxy" | tr -d ' ')
  [ "$d" -eq 0 ] && return
  if [ "$d" -eq "$p" ]; then
    echo -e "    ${GREEN}↳ match (${d}B)${NC}"
  else
    local pct=$(( (p - d) * 100 / d ))
    echo -e "    ${RED}↳ MISMATCH direct=${d} proxy=${p} (${pct}%) — binary corruption${NC}"
  fi
}

slug() { echo "$1" | tr ' ()/' '____' | tr -cd 'a-zA-Z0-9_-'; }

# ── Payload ──────────────────────────────────────────────────

PAYLOAD='{
  "schoolProfile": {
    "schoolName":"Smoke Test School","state":"TX","schoolType":"microschool",
    "entityType":"llc_single","schoolStage":"new_school",
    "fundingProfile":"tuition_based","openingYear":2026,"maxCapacity":25,
    "fiscalYearStartMonth":7,"isPartialFirstYear":true,
    "year1OperatingMonths":10,"debtIncluded":false
  },
  "enrollment":{"year1":12,"year2":18,"year3":22,"year4":25,"year5":25},
  "revenueRows":[
    {"id":"r1","category":"tuition_and_fees","lineItem":"Tuition","enabled":true,
     "driverType":"per_student","amounts":[10000,10300,10609,10927,11255]}
  ],
  "staffingRows":[
    {"id":"s1","roleName":"Teacher","functionCategory":"instructional",
     "employmentType":"full_time","fte":1,"annualizedRate":45000,
     "benefitsEligible":true,"benefitsRate":20,"payrollTaxRate":7.65,
     "payrollLike":false,"notes":""}
  ],
  "expenseRows":[
    {"id":"e1","category":"occupancy_facility","lineItem":"Rent","enabled":true,
     "driverType":"monthly","amounts":[2500,2575,2652,2732,2814]}
  ],
  "capitalAndDebtRows":[]
}'

# ── Header ───────────────────────────────────────────────────

echo ""
echo -e "${BOLD}╔══════════════════════════════════════════════════════════════╗${NC}"
echo -e "${BOLD}║         SchoolStack Budget — Export Smoke Test              ║${NC}"
echo -e "${BOLD}╚══════════════════════════════════════════════════════════════╝${NC}"
echo ""
echo "  Netlify:  $NETLIFY_BASE"
echo "  Railway:  $RAILWAY_BASE"
echo "  Output:   $OUT_DIR"
echo ""

# ── PUBLIC XLSX (3 routes) ───────────────────────────────────

PUB_XLSX=(
  "public/export-budget:Budget"
  "public/export-underwriting:Underwriting"
  "public/export-single-year:Single-Year"
)

for tl in "Netlify:$NETLIFY_BASE" "Railway:$RAILWAY_BASE"; do
  t="${tl%%:*}"; b="${tl#*:}"
  echo -e "${BOLD}── Public XLSX ($t) ──${NC}"
  echo ""
  for entry in "${PUB_XLSX[@]}"; do
    route="${entry%%:*}"; label="${entry#*:}"
    s=$(slug "${label}_${t}")
    curl -s -X POST "$b/api/$route" \
      -H "Content-Type: application/json" -d "$PAYLOAD" \
      -o "$OUT_DIR/pub_${s}.xlsx" 2>/dev/null
    check_xlsx "$OUT_DIR/pub_${s}.xlsx" "$label ($t)"
  done
  echo ""
done

echo -e "${CYAN}  Public size comparison:${NC}"
for entry in "${PUB_XLSX[@]}"; do
  label="${entry#*:}"
  compare_sizes \
    "$OUT_DIR/pub_$(slug "${label}_Railway").xlsx" \
    "$OUT_DIR/pub_$(slug "${label}_Netlify").xlsx"
done
echo ""

# ── AUTHENTICATED ROUTES ─────────────────────────────────────

if [ -z "$TOKEN" ] || [ -z "$MODEL_ID" ]; then
  echo -e "${YELLOW}  ⏭  Skipping auth routes (no TOKEN / MODEL_ID)${NC}"
  echo "     Usage: $0 <JWT_TOKEN> <MODEL_ID>"
  echo "     Get token: DevTools > Application > localStorage > auth_token"
  echo ""; SKIP=1
else
  AUTH="Authorization: Bearer $TOKEN"

  # ── 5 XLSX ──

  AUTH_XLSX=(
    "export:Standard"
    "export/lender-proforma:Lender Pro Forma"
    "export/underwriting:Underwriting V1 (legacy)"
    "export/underwriting-v2:Underwriting V2 (21-tab)"
    "export/single-year:Single-Year"
  )

  for tl in "Netlify:$NETLIFY_BASE" "Railway:$RAILWAY_BASE"; do
    t="${tl%%:*}"; b="${tl#*:}"
    echo -e "${BOLD}── Auth XLSX ($t) ──${NC}"
    echo ""
    for entry in "${AUTH_XLSX[@]}"; do
      route="${entry%%:*}"; label="${entry#*:}"
      s=$(slug "${label}_${t}")
      curl -s -H "$AUTH" "$b/api/models/$MODEL_ID/$route" \
        -o "$OUT_DIR/auth_${s}.xlsx" 2>/dev/null
      check_xlsx "$OUT_DIR/auth_${s}.xlsx" "$label ($t)"
    done
    echo ""
  done

  echo -e "${CYAN}  Auth XLSX size comparison:${NC}"
  for entry in "${AUTH_XLSX[@]}"; do
    label="${entry#*:}"
    compare_sizes \
      "$OUT_DIR/auth_$(slug "${label}_Railway").xlsx" \
      "$OUT_DIR/auth_$(slug "${label}_Netlify").xlsx"
  done
  echo ""

  # ── 4 PDF ──

  AUTH_PDF=(
    "export/pro-forma-pdf:Pro Forma PDF"
    "export/loan-readiness-pdf:Loan Readiness PDF"
    "export/lender-packet-pdf:Lender Packet PDF"
    "export/board-packet-pdf:Board Packet PDF"
  )

  for tl in "Netlify:$NETLIFY_BASE" "Railway:$RAILWAY_BASE"; do
    t="${tl%%:*}"; b="${tl#*:}"
    echo -e "${BOLD}── Auth PDF ($t) ──${NC}"
    echo ""
    for entry in "${AUTH_PDF[@]}"; do
      route="${entry%%:*}"; label="${entry#*:}"
      s=$(slug "${label}_${t}")
      curl -s -H "$AUTH" "$b/api/models/$MODEL_ID/$route" \
        -o "$OUT_DIR/auth_${s}.pdf" 2>/dev/null
      check_pdf "$OUT_DIR/auth_${s}.pdf" "$label ($t)"
    done
    echo ""
  done

  echo -e "${CYAN}  Auth PDF size comparison:${NC}"
  for entry in "${AUTH_PDF[@]}"; do
    label="${entry#*:}"
    compare_sizes \
      "$OUT_DIR/auth_$(slug "${label}_Railway").pdf" \
      "$OUT_DIR/auth_$(slug "${label}_Netlify").pdf"
  done
  echo ""

  # ── 2 JSON ──

  AUTH_JSON=(
    "export/lender-packet:Lender Packet JSON"
    "export/board-packet:Board Packet JSON"
  )

  for tl in "Netlify:$NETLIFY_BASE" "Railway:$RAILWAY_BASE"; do
    t="${tl%%:*}"; b="${tl#*:}"
    echo -e "${BOLD}── Auth JSON ($t) ──${NC}"
    echo ""
    for entry in "${AUTH_JSON[@]}"; do
      route="${entry%%:*}"; label="${entry#*:}"
      s=$(slug "${label}_${t}")
      curl -s -H "$AUTH" "$b/api/models/$MODEL_ID/$route" \
        -o "$OUT_DIR/auth_${s}.json" 2>/dev/null
      check_json "$OUT_DIR/auth_${s}.json" "$label ($t)"
    done
    echo ""
  done
fi

# ── SUMMARY ──────────────────────────────────────────────────

TOTAL=$((PASS + FAIL))
echo -e "${BOLD}╔══════════════════════════════════════════════════════════════╗${NC}"
echo -e "${BOLD}║                         RESULTS                            ║${NC}"
echo -e "${BOLD}╚══════════════════════════════════════════════════════════════╝${NC}"
echo ""
echo -e "  ${GREEN}Passed: $PASS${NC}  |  ${RED}Failed: $FAIL${NC}  |  Total: $TOTAL"
[ "$SKIP" -gt 0 ] && echo -e "  ${YELLOW}(Auth routes skipped — provide TOKEN MODEL_ID)${NC}"
echo ""
echo "  Files: $OUT_DIR/"
echo ""

if [ "$FAIL" -gt 0 ]; then
  # Diagnosis
  nf=0; df=0; nh=0
  for f in "$OUT_DIR"/*_Netlify*; do
    [ -f "$f" ] || continue
    if is_html "$f"; then nh=$((nh+1))
    elif [ "$(wc -c < "$f" | tr -d ' ')" -lt 100 ]; then nf=$((nf+1)); fi
  done
  for f in "$OUT_DIR"/*_Railway*; do
    [ -f "$f" ] || continue
    [ "$(wc -c < "$f" | tr -d ' ')" -lt 100 ] && df=$((df+1))
  done

  echo -e "  ${BOLD}Diagnosis:${NC}"
  if [ "$nh" -gt 0 ]; then
    echo -e "  ${RED}→ Netlify serving HTML for API routes — _redirects SPA catch-all is intercepting /api/*${NC}"
    echo "    Fix: Ensure /api/* proxy rule comes before /* catch-all in _redirects"
  elif [ "$df" -eq 0 ] && [ "$nf" -gt 0 ]; then
    echo -e "  ${RED}→ Railway OK, Netlify corrupt — proxy encoding issue${NC}"
    echo "    Fix: Add Content-Length + use res.end(Buffer.from(buffer)) in models.ts"
  elif [ "$df" -gt 0 ]; then
    echo -e "  ${RED}→ Railway failing — server-side export error${NC}"
    echo "    Check Railway logs for error details"
  else
    echo -e "  ${RED}→ Binary corruption detected — check size comparison above${NC}"
    echo "    Fix: Add Content-Length + use res.end(Buffer.from(buffer)) in models.ts"
  fi
  echo ""
  exit 1
else
  echo -e "  ${GREEN}${BOLD}✅ All exports valid.${NC}"
  echo ""
  exit 0
fi
